package t6Exercise;

import java.util.Scanner;

public class P08TriangleofNumbers {

	public static void main(String[] args) {

		Scanner scan = new Scanner(System.in);
		int a = Integer.parseInt(scan.nextLine());
		int num =1;
		for(int i=1;i<=a;++i) {
			for(int j=1;j<=i;++j) {
				
				System.out.printf("%d ",num);
			}
			num=num+1;
			System.out.println();
		}
	}

}
